// Indexes & Constraints Setup
CREATE CONSTRAINT candidate_id ON (c:Candidate) ASSERT c.candidateId IS UNIQUE;
CREATE CONSTRAINT job_id ON (j:Job) ASSERT j.jobId IS UNIQUE;
CREATE CONSTRAINT skill_name ON (s:Skill) ASSERT s.name IS UNIQUE;
CREATE INDEX candidate_location FOR (c:Candidate) ON (c.locationId);
CREATE INDEX job_status FOR (j:Job) ON (j.status, j.postedAt);
CREATE FULLTEXT INDEX skill_search FOR (n:Skill) ON EACH [n.name, n.category];

--- TABLE ---
// Candidate Matching Query
MATCH (j:Job {jobId: $jobId})-[:REQUIRES_SKILL]->(s:Skill)
WITH j, collect(s) AS required
MATCH (c:Candidate)-[r:HAS_SKILL]->(s)
WHERE s IN required AND c.isLooking = true
WITH c, count(s) AS matched, size(required) AS total,
     sum(r.proficiency) AS totalProf
RETURN c.candidateId, c.name,
       (matched * 1.0 / total) AS matchScore,
       totalProf
ORDER BY matchScore DESC, totalProf DESC
LIMIT 20;

--- TABLE ---
// Skill Gap Analysis
MATCH (j:Job {jobId: $jobId})-[:REQUIRES_SKILL]->(s:Skill)
WHERE NOT (:Candidate {candidateId: $candidateId})-[:HAS_SKILL]->(s)
OPTIONAL MATCH (s2:Skill)-[:RELATED_TO]->(s)
WHERE (:Candidate {candidateId: $candidateId})-[:HAS_SKILL]->(s2)
RETURN s.name AS missingSkill,
       s.category AS category,
       s2.name AS adjacentSkillYouHave;

--- TABLE ---
// Referral Path (Up to 3 Degrees)
MATCH path = shortestPath(
  (me:Candidate {candidateId: $myId})-[:CONNECTED_TO*1..3]-
  (insider:Candidate)-[:WORKED_AT]->(c:Company {companyId: $targetCompanyId})
)
RETURN [n IN nodes(path) | n.name] AS connectionPath,
       length(path) AS degrees
ORDER BY degrees
LIMIT 5;

--- TABLE ---
// Trending Skills (Last 30 Days)
MATCH (j:Job)-[r:REQUIRES_SKILL]->(s:Skill)
WHERE j.postedAt > datetime() - duration('P30D')
  AND j.status = 'active'
RETURN s.name, s.category, count(*) AS demand
ORDER BY demand DESC
LIMIT 20;

What is Vibe Coding?
Vibe Coding means iterating quickly with AI-assisted development — writing prompts to generate features, then refining the output. This section provides the exact prompts, patterns, and architecture decisions to make vibe coding WorkVerse as smooth as possible.

--- TABLE ---
Layer | Tech | Why
Database | Neo4j AuraDB (free tier for dev) | Managed graph DB, instant Cypher access
Backend | Node.js + Express or FastAPI (Python) | Great Neo4j driver support
ORM / Query Builder | neogma (Node) or neomodel (Python) | Schema-first graph modelling
Frontend | React + Vite + TailwindCSS | Fast iteration, rich UI component ecosystem
Graph Viz | React-Force-Graph or Neovis.js | Native Neo4j visualisation in browser
Auth | Auth0 / Clerk | Plug-and-play, JWT for API auth
Search | Neo4j Fulltext Index + Algolia (optional) | Fulltext on top of graph for speed
Hosting | Vercel (frontend) + Railway / Render (backend) | Zero-config deployment, git push to deploy
Testing | Jest + Supertest + Neo4j Test Container | Graph-aware unit + integration tests
